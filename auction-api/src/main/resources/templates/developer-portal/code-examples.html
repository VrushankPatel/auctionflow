Python
from auctionflow_sdk import AuctionFlowClient

client = AuctionFlowClient("https://api.auctionflow.com")
client.login(AuthRequest(email="user@example.com", password="password"))

auction = CreateAuctionRequest(
    itemId="item-123",
    categoryId="electronics",
    auctionType="ENGLISH_OPEN",
    reservePrice=100.0,
    buyNowPrice=500.0,
    startTime="2023-10-01T10:00:00Z",
    endTime="2023-10-01T12:00:00Z"
)
client.create_auction(auction)
cURL
curl -X POST "https://api.auctionflow.com/api/v1/auctions" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "itemId": "item-123",
    "categoryId": "electronics",
    "auctionType": "ENGLISH_OPEN",
    "reservePrice": 100.00,
    "buyNowPrice": 500.00,
    "startTime": "2023-10-01T10:00:00Z",
    "endTime": "2023-10-01T12:00:00Z"
  }'

from auctionflow_sdk import AuctionFlowClient

client = AuctionFlowClient("https://api.auctionflow.com")

cURL
curl -X POST "https://api.auctionflow.com/api/v1/auctions/auction-123/bids" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"amount": 150.00}'

Python
from auctionflow_sdk import AuctionFlowClient

client = AuctionFlowClient("https://api.auctionflow.com")
client.login(AuthRequest(email="user@example.com", password="password"))

auctions = client.list_auctions(category="electronics", page=0, size=10)
for auction in auctions:
    print(f"Auction: {auction.title}, Current Bid: {auction.currentBid}")
cURL
curl -X GET "https://api.auctionflow.com/api/v1/auctions?category=electronics&page=0&size=10" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
